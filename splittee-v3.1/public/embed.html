🔧 Integration Code

This is how Pinewood Golf Club integrated Split Tee. Copy this pattern for your course:

<!-- 1. Add the Split Tee SDK -->
<script src="https://js.splittee.com/v1.js"></script>

<!-- 2. Initialize with your API key -->
<script>
  SplitTee.init({
    apiKey: 'sk_live_your_api_key',
    onSuccess: function(split) {
      console.log('Split created!', split.shortCode);
    },
    onError: function(error) {
      console.error('Split failed:', error);
    }
  });
</script>

<!-- 3. Trigger split payment -->
<button onclick="openSplitPayment()">Split with Group</button>

<script>
  function openSplitPayment() {
    SplitTee.createSplit({
      amount: 35000,        // $350.00 in cents
      players: 4,
      teeTime: '2025-04-20T09:00:00',
      courseName: 'Pinewood Golf Club',
      bookerEmail: 'john@example.com',
      metadata: { bookingId: 'BK-12345' }
    });
  }
</script>
// Server-side: Create split via API
const response = await fetch('https://api.splittee.com/v1/splits', {
  method: 'POST',
  headers: {
    'X-SplitTee-Key': process.env.SPLITTEE_API_KEY,
    'Content-Type': 'application/json'
  },
  body: JSON.stringify({
    amount: 35000,
    players: [
      { email: 'player1@example.com', name: 'John' },
      { email: 'player2@example.com', name: 'Mike' },
      { email: 'player3@example.com', name: 'Sarah' },
      { email: 'player4@example.com', name: 'Lisa' }
    ],
    teeTime: '2025-04-20T09:00:00Z',
    webhookUrl: 'https://yourcourse.com/webhooks/splittee',
    metadata: { bookingId: 'BK-12345' }
  })
});

const split = await response.json();

// Returns payment links for each player
console.log(split.players);
// [
//   { email: 'player1@...', paymentUrl: 'https://pay.splittee.com/abc123' },
//   { email: 'player2@...', paymentUrl: 'https://pay.splittee.com/def456' },
//   ...
// ]
